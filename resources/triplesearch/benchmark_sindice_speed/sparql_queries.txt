# 1 query per line. Query must be written in SPARQL


#1. Give me all instances of "<http://dbpedia.org/ontology/Artist>"
PREFIX dbpedia-owl: <http://dbpedia.org/ontology/> PREFIX rdfs: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> SELECT DISTINCT ?s WHERE{ ?s rdfs:type <http://dbpedia.org/ontology/Artist>.}

#2. Give me all instances of "dbpedia-ont:Organisation"
PREFIX dbpedia-owl: <http://dbpedia.org/ontology/> PREFIX rdfs: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> SELECT DISTINCT ?s WHERE{ ?s rdfs:type <http://dbpedia.org/ontology/Organisation>.}

#3. Give me all parents of "<..>pig"
PREFIX dbpedia-owl: <http://dbpedia.org/ontology/> PREFIX rdfs: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> SELECT DISTINCT ?o WHERE{ <http://dbpedia.org/resource/Pig>  rdfs:type ?o.}


#4. Give me all athletes who are chinese
PREFIX dbpedia-owl: <http://dbpedia.org/ontology/> PREFIX rdfs: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> PREFIX dbpp: <http://dbpedia.org/property/> SELECT DISTINCT ?s WHERE{ ?s  rdfs:type dbpedia-owl:Athlete; dbpp:nationality <http://dbpedia.org/resource/China>}

get me all datasets in sindice

PREFIX dataset: <http://vocab.sindice.net/dataset/1.0/> 
     SELECT ?dataset_uri ?dataset_name ?dataset_domain
     FROM <http://sindice.com/dataspace/default/dataset/index> 
     WHERE {
        ?s dataset:uri ?dataset_uri .
        ?s dataset:name ?dataset_name.
        ?s dataset:domain ?dataset_domain.
     }
